    }

    if (user.blockedUsers.includes(targetUserId)) {
      throw Error("Impossible d'accepter l'amiti, l'utilisateur cible est bloqu.");
    }
 
    // Rechercher l'ID du targetUserId dans la liste des amis de l'utilisateur actuel
